export const AnnotatedScreenshot = ({ desktopSrc, mobileSrc, callouts = [] }) => {
  const p = 'as' + (desktopSrc || '').replace(/[^a-z0-9]/gi, '').slice(-10);
  const desktopCallouts = callouts.filter(c => !c.target || c.target === 'desktop');
  const mobileCallouts = callouts.filter(c => c.target === 'mobile');

  const css = [
    `.${p}-panel{display:none}`,
    ...callouts.map(c => `#${p}-c${c.id}:checked~.${p}-panels>.${p}-p${c.id}{display:flex}`),
  ].join('');

  const dot = (c) => ({
    position: 'absolute',
    top: c.y,
    left: c.x,
    width: '26px',
    height: '26px',
    borderRadius: '50%',
    background: '#F97316',
    color: '#fff',
    fontWeight: '700',
    fontSize: '12px',
    lineHeight: 1,
    display: 'flex',
    alignItems: 'center',
    justifyContent: 'center',
    cursor: 'pointer',
    transform: 'translate(-50%,-50%)',
    boxShadow: '0 2px 8px rgba(249,115,22,0.45)',
    userSelect: 'none',
    zIndex: 10,
    border: '2px solid rgba(255,255,255,0.9)',
  });

  return (
    <div style={{
      borderRadius: '12px',
      overflow: 'hidden',
      boxShadow: '0 1px 3px rgba(0,0,0,0.07), 0 6px 24px rgba(0,0,0,0.08)',
      background: '#ffffff',
      marginBottom: '24px',
    }}>
      <style>{css}</style>

      {callouts.map(c => (
        <input key={c.id} type="radio" name={p} id={`${p}-c${c.id}`} style={{ display: 'none' }} />
      ))}

      <div style={{
        display: 'flex',
        gap: '12px',
        background: '#f4f5f7',
        padding: '16px',
        alignItems: 'flex-start',
      }}>
        <div style={{ position: 'relative', flex: mobileSrc ? '0 0 calc(60% - 6px)' : '1' }}>
          <img
            src={desktopSrc}
            alt=""
            style={{ width: '100%', display: 'block', borderRadius: '6px' }}
          />
          {desktopCallouts.map(c => (
            <label key={c.id} htmlFor={`${p}-c${c.id}`} style={dot(c)}>
              {c.id}
            </label>
          ))}
        </div>

        {mobileSrc && (
          <div style={{ position: 'relative', flex: '0 0 calc(40% - 6px)' }}>
            <img
              src={mobileSrc}
              alt=""
              style={{ width: '100%', display: 'block', borderRadius: '6px' }}
            />
            {mobileCallouts.map(c => (
              <label key={c.id} htmlFor={`${p}-c${c.id}`} style={dot(c)}>
                {c.id}
              </label>
            ))}
          </div>
        )}
      </div>

      <div className={`${p}-panels`}>
        {callouts.map(c => (
          <div
            key={c.id}
            className={`${p}-panel ${p}-p${c.id}`}
            style={{
              padding: '14px 20px',
              borderTop: '1px solid #f0f0f0',
              alignItems: 'flex-start',
              gap: '12px',
            }}
          >
            <div style={{
              width: '26px',
              height: '26px',
              borderRadius: '50%',
              background: '#F97316',
              color: '#fff',
              fontWeight: '700',
              fontSize: '12px',
              lineHeight: 1,
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',
              flexShrink: 0,
            }}>
              {c.id}
            </div>
            <div>
              <div style={{ fontWeight: '600', fontSize: '14px', color: '#111827', marginBottom: '4px' }}>
                {c.title}
              </div>
              <div style={{ color: '#6b7280', fontSize: '13px', lineHeight: '1.6' }}>
                {c.body}
              </div>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
};
